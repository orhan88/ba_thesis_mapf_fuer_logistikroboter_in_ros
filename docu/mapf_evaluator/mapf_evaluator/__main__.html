 55class Evaluator(Node):
 56    '''
 57    This class represents the following ros2 node: **mapf_evaluator_node**.  
 58
 59    The node mapf_evaluator_node subscribes to the folllowing topic: **/mapf_statistics_info**  
 60    Every time a message on the topic is received, the following function will be called: **callback(self, msg)**
 61
 62    The evaluator is executed by the function **run()**. This function creates a matplotlib.pyplot figure and 
 63    then runs a while loop as long as the node mapf_evaluator_node is runnning.  
 64    Every loop cycle the ros2 node is spinned once and the figure is updated with new data
 65    '''
 66
 67    def __init__(self):        
 68        super().__init__('mapf_evaluator_node')
 69        self.subscriber = self.create_subscription(StatisticInfo, 'mapf_statistics_info', self.callback, 10)
 70        '''**rclpy.subscription.Subscription** - ros2 subscriber to the topic */mapf_statistics_info*'''
 71        self.algorithms = []  
 72        '''**list** - list of saved algorithms. Items are objects of the class "Algorithm" '''
 73        #list of saved steps
 74        #self.step = []  
 75        #'''list - list of saved steps. Items are type of the class "Algorithm" '''
 76        #number defines how many algorithms stored, max 4.
 77        self.number = 0
 78        '''**int** - describes the number of saved algorithms'''
 79
 80        #color combinations: (line / marker)
 81        #first:  -> green / blue
 82        #second: -> red   / yello
 83        #third:  -> black / white
 84        #last:   -> cyan  / magenta
 85        self.colors = ["g", "r", "k", "c"]
 86        '''
 87        **list** - list of colors for plotting the data.
 88
 89        Per default the first algorithm will be plotted in green, second in red, third in black and the last in cyan'''
 90        self.markerfacecolors = ["b", "y", "w", "m"]
 91        '''
 92        **list** - list of colors for plotting the markerpoints
 93        
 94        Per default the markerpoints for the first algorithm will be plotted in blue, for the second in yellow, third in white and the last in magenta
 95        '''
 96        
 97    def callback(self, msg):
 98        '''
 99        This function will be called every time a message on the topic /mapf_statistics_info is received.  
100        It saves the received message by finding the right algorithm from self.algorithms and then saving it in the right spot.  
101        
102        If no message for the algorithm has been saved so far, then it creates an instance of the class "Algorithm".
103        This instance will be saved in self.algorithms, so next messages can be added later here.
104
105        For reasons of clarity the max number of 4 different algorithms can be saved and plotted by the evaluator.
106        If the number exceeds 4, then the first added algorithm will be deleted from self.algorithms
107        
108        :param msg: mapf_interfaces/StatisticInfo.msg - message from the topic /mapf_statistics_info
109        '''
110        self.get_logger().info('New callback received. Algorithm name: "%s", STEP: %s, SoC: %f, Runtime: %f ' % (msg.name_algorithm, msg.step, msg.sum_of_cost, msg.runtime))
111        #add new algorithm to the list if no message received for this algorithm
112        if self.check_new_algorithm(msg.name_algorithm):
113            self.algorithms.append(Algorithm(msg.name_algorithm, self.colors[self.number%4], self.markerfacecolors[self.number%4]))
114            self.number += 1
115            #dont store more than four algorithms
116            if self.number > 4:
117                #remove first element from the list
118                self.algorithms.pop(0)
119        self.save_msg(msg)
120        #self.print_all_algorithms()
121
122    def save_msg(self, msg):
123        '''
124        This function saves the received msg in self.algorithm. It iterates the list of saved algorithms till it finds the algorithm with the right name. 
125        After it saves the message for the right algorithm
126
127        :param msg: mapf_interfaces/StatisticInfo.msg - message from the topic /mapf_statistics_info
128        '''
129        #save the message for the correct algorithm 
130        for algorithm in self.algorithms:
131            if msg.name_algorithm == algorithm.name:
132                algorithm.msg.append(msg)
133                return
134
135    def check_new_algorithm(self, name):
136        '''
137        This function checks whether there is already an instance of the class Algorithm saved in self._algorithms with the given name
138        
139        :param name: string - name of the algorithm
140        
141        :return: True, if an instance exists with the given name  
142        False, if there is no instance with the given name
143        '''
144        for algorithm in self.algorithms:
145            if algorithm.name == name:
146                return False
147        return True
148
149    def print_all_algorithms(self):
150        '''
151        Help function for printing all the saved data
152        '''
153        print(len(self.algorithms))
154        for algorithm in self.algorithms:
155            print(algorithm.name)
156            print(algorithm.color)
157            print(algorithm.markerfacecolor)
158            print(len(algorithm.msg))
159
160    def run(self):
161        '''
162        Executes the evaluator
163
164        Creates a matplotlib.pyplot figure and then runs a while loop as long as the ros2 node mapf_evaluator_node is runnning.  
165        Every loop cycle the ros2 node is spinned once and the figure is updated with new data.
166        '''
167        plt.ion()
168        plt.show()
169        plt.figure(num='Algorithm evaluation')        
170
171        while(rclpy.ok()):
172        
173            plt.clf()
174            #plot "sum of cost" over "step" for every algorithm
175            for algorithm in self.algorithms:
176                step = []
177                sof = []
178                runtime =[]
179                #for every message saved for this algorithm
180                for msg in algorithm.msg:
181                    #get step, sum_of_cost and runtime from msg
182                    step.append(msg.step)
183                    sof.append(msg.sum_of_cost)
184                    runtime.append(msg.runtime)
185
186                #print(step)    
187                #print(sof)    
188                #plotting sum of cost over step
189                plot1 = plt.subplot(2, 1, 1)
190                plot1.plot(step, sof, label = algorithm.name, color=algorithm.color, linestyle='dashed', linewidth = 3,
191                        marker='o', markerfacecolor=algorithm.markerfacecolor, markersize=12)
192                plot1.legend()
193                # naming the x axis    
194                plt.xlabel('STEP')
195                # naming the y axis
196                plt.ylabel('SUM OF COST')
197                #plt.title('Sum of cost')
198                plot1.set_xlim(left=0)
199                plot1.set_ylim(bottom=0)
200                plot1.autoscale(True)
201                
202                plot2 = plt.subplot(2, 1, 2)
203                plot2.plot(step, runtime, label = algorithm.name, color=algorithm.color, linestyle='dashed', linewidth = 3,
204                        marker='o', markerfacecolor=algorithm.markerfacecolor, markersize=12)
205                plot2.legend()
206                plot2.set_xlim(left=0)
207                plot2.set_ylim(bottom=0)
208                plot2.autoscale(True)
209                # naming the x axis    
210                plt.xlabel('STEP')
211                # naming the y axis
212                plt.ylabel('RUNTIME')
213            
214            #update figure
215            plt.draw()
216            plt.pause(0.1)
217            
218            #update ROS            
219            rclpy.spin_once(self, timeout_sec=0.0)
220            
221def main(args = None):
222    '''
223        Main function: creates the Evaluator and executes it.
224    '''
225    rclpy.init(args=args)
226
227    evaluator = Evaluator()
228    evaluator.run()
229
230    evaluator.destroy_node()
231    rclpy.shutdown()

235    sys.exit(main())

It saves the received message by finding the right algorithm from self.algorithms and then saving it in the right spot.

If no message for the algorithm has been saved so far, then it creates an instance of the class "Algorithm". This instance will be saved in self.algorithms, so next messages can be added later here.
