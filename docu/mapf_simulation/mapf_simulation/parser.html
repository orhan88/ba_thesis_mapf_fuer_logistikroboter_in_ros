 36    def load_map(self, filename):
 37        '''
 38        This function is used for loading the .map file.  
 39
 40        The function checks first whether the file exists and can be opened. 
 41        Then it tries to read from the file. 
 42        After reading the file the function analyzes the content and the syntax.
 43        If the test is positive then it loads the data into the model!
 44
 45        :param filename: string - the complete path to the file including the name of the file
 46
 47        :return:    1, if the file was loaded successfully
 48                    0, otherwise
 49        '''
 50        if self._model is None:
 51            return 0
 52        if self._model_view is None:
 53            return 0
 54
 55        #CHECK IF THE FILE CAN BE OPENED
 56        if not os.path.isfile(filename):
 57            print('ERROR: CAN NOT OPEN THE FILE: ', filename)
 58            return 0
 59
 60        #TRY TO READ THE FILE
 61        print('\nSTATUS: Loading the map file: ' + filename)
 62        try:
 63            ff = open(filename)
 64            lines = ff.read().splitlines()
 65            ff.close()
 66        except RuntimeError as error:
 67            print('ERROR: RUNTIME EXCEPTION WHILE LOADING THE MAP FILE: ' + filename)
 68            print(error)
 69            return 0
 70
 71        #ANALYZE THE CONTENT OF THE FILE
 72        if not self.analyze_file_map(lines):
 73            print('ERROR: WRONG SYNTAX, SKIP THE .MAP FILE!')
 74            return 0    
 75
 76        #UPDATE MODEL: IMPORT DATA 
 77        self._model.mapf_update_map(os.path.basename(os.path.normpath(filename)), 
 78                                    self._size_x,
 79                                    self._size_y,
 80                                    lines[4:])
 81        
 82        #UPDATE MODELVIEW: IMPORT DATA + RESIZE TO FIT THE WINDOW
 83        self._model_view.update()
 84        self._model_view.resize_to_fit()
 85
 86        print('STATUS: MAP FILE LOADED SUCCESSFULLY')

451    def check_collisions(self, robot_nodes, goal_nodes):
452        '''
453        This function checks if there are conflicts between the different 
454        items in the map. 
455        
456        A conflict occurs in the following cases:
457        -   two or more agents are trying to occupy the same spot
458        -   two or more robots have the same target
459        -   the start position of an agent collides with a blocked node in the map
460        -   the target spot of an agent collides with a blocked node in the map
461
462        :param robot_nodes: list of (x,y) - where every entry (x,y) represents the start position for one robot
463        :param goal_nodes: list of (x,y) - where every entry (x,y) represents the end position for one robot
464        :return: **True**, if no conflict was found  
465        **False**, if a conflict was found
466        '''
467
468        #Note: in the model position counting stars by (1,1) 
469        #      and in the scene file by (0,0)
470        blocked_nodes = self._model.get_blocked_nodes()
471        
472        #1. TEST IF TWO AGENTS TRY TO OCCUPY THE SAME SPOT
473        newlist = []
474        line = 2
475        for node in robot_nodes:
476            
477            if node not in newlist:
478                newlist.append(node)        
479            else:
480                #Attention: node[0]-1, because in the model position counting stars by (1,1) 
481                #                      and in the scene file by (0,0)
482                print('Collision occured in the map between two agents, start positions are equal, line: ', line,
483                       "\nConflict at position: ", str(node[0]-1) + " " + str(node[1]-1),
484                       "\nTwo agents cannot have the same start position!")
485                return True    
486            
487            line=line+1
488
489        #2. TEST IF TWO GOAL NODES OCCUPY THE SAME SPOT
490        newlist = []
491        line = 2
492        for node in goal_nodes:
493            
494            if node not in newlist:
495                newlist.append(node)        
496            else:
497                print('Collision occured in the map, two goal positions are equal, line: ', line,
498                       "\nConflict at goal position: ", str(node[0]-1) + " " + str(node[1]-1),
499                       "\nTwo agents cannot have same goal position!")
500                return True    
501            
502            line=line+1
503          
504        #3. TEST IF A ROBOT START POSITION COLLIDES WITH A BLOCKED NODE IN THE MAP
505        line = 2
506        for node in robot_nodes:
507            if node in blocked_nodes:
508                print('Collision occured in the map, start position of an agent is trying to occupy a blocked spot, '+
509                      "line: " + str(line) + " in the .scen file"
510                      "\nConflict at position: x: " + str(node[0]-1) + ", y:  " + str(node[1]-1) + 
511                      ", position is already occupied!")
512                return True    
513            line=line+1
514        
515        #4. TEST IF A ROBOT GOAL POSITION COLLIDES WITH A BLOCKED NODE IN THE MAP
516        line = 2
517        for node in goal_nodes:
518            if node in blocked_nodes:
519                print('Collision occured in the map, goal position of an agent is a blocked spot, '+
520                      "line: " + str(line) + " in the .scen file"
521                      "\nConflict at position: x: " + str(node[0]-1) + ", y:  " + str(node[1]-1) + 
522                      ", position is already occupied!")
523                return True    
524            line=line+1
525          
526        return False

This function checks if there are conflicts between the different items in the map.

A conflict occurs in the following cases:

  • two or more agents are trying to occupy the same spot
  • two or more robots have the same target
  • the start position of an agent collides with a blocked node in the map
  • the target spot of an agent collides with a blocked node in the map
Parameters
  • robot_nodes: list of (x,y) - where every entry (x,y) represents the start position for one robot
  • goal_nodes: list of (x,y) - where every entry (x,y) represents the end position for one robot
Returns

True, if no conflict was found
False, if a conflict was found
